add_executable(
    test_output
        test_output.cpp
    )
target_include_directories(
    test_output
    PUBLIC ${Boost_INCLUDE_DIR}             
           ../../outfile/include
    )

    test_output
        ${Boost_LIBRARIES} 
        swmm-output
    )

    PROPERTIES 
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
